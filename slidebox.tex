% slidebox.tex
% PDF styling for the article-style boxes (see slidebox.lua).
% Included via _quarto.yml -> format: pdf: include-in-header, AFTER
% pdf-theme.tex (which defines the sl* Blueprint colours used below).

\usepackage{tcolorbox}
\tcbuselibrary{skins,breakable}

% leading-zero helper for the slidebox counter (01, 02, ... like the site)
\makeatletter
\newcommand\sbtwodigits[1]{\ifnum#1<10 0\fi\number#1}
\makeatother

% A slidebox: white card, thin border, a blue accent rule down the LEFT
% edge, a faint two-digit counter in the top-right corner, bold navy sans
% title, and an optional lower part (\tcblower) holding the footer in
% IBM Plex Mono / accent-blue, divided by a thin rule. Mirrors the
% .slidebox component in ember.scss.
\newtcolorbox[auto counter]{slidebox}[1]{%
  enhanced, breakable,
  colback=white,
  colframe=slborder,
  boxrule=0.4pt,
  arc=2pt,
  sharp corners=west,                     % square left edge for the accent rule
  borderline west={2.2pt}{0pt}{slaccent}, % blue spine
  left=4mm, right=4mm, top=2.4mm, bottom=2.4mm,
  before skip=10pt, after skip=10pt,
  title={#1},
  halign title=left,
  colbacktitle=white,
  coltitle=slhead,
  fonttitle=\sffamily\bfseries\large,
  toptitle=1mm, bottomtitle=1.4mm,
  lower separated=true,
  collower=slaccentdk,
  fontlower=\ttfamily\small,
  segmentation style={solid, slrule, line width=0.4pt},
  % faint leading-zero counter, top-right (first piece only)
  overlay unbroken and first={%
    \node[anchor=north east, inner sep=0pt,
          font=\ttfamily\scriptsize\color{slfaint}]
      at ([xshift=-3.2mm, yshift=-2.2mm]frame.north east)
      {\sbtwodigits{\thetcbcounter}};%
  },
}

% Kicker label (Goals / Key Equation / Key Concept ...): a small bordered
% mono badge in accent-blue, mirroring the .kicker badge in ember.scss.
\newcommand{\kicker}[1]{%
  {\setlength{\fboxsep}{3.5pt}\setlength{\fboxrule}{0.4pt}%
   \fcolorbox{slborder}{white}{%
     \ttfamily\footnotesize\color{slaccentdk}\MakeUppercase{#1}}}%
  \par\vspace{5pt}%
}
