% pdf-theme.tex
% Global Blueprint theming for PDF output — mirrors ember.scss (the website).
% Included first via _quarto.yml -> format: pdf: include-in-header.
% Engine-agnostic: the plex-* packages load Type1 under pdflatex and the
% OpenType faces under xelatex/lualatex.

% ── Fonts: IBM Plex (matches the site) ───────────────────────────
\usepackage[sfdefault]{plex-sans}   % body + headings -> IBM Plex Sans
\usepackage{plex-mono}              % \texttt / verbatim -> IBM Plex Mono
% (math is left at the class default for legibility)

% ── Ink/Coral palette (same hex values as ember.scss) ────────────
\definecolor{slhead}{HTML}{1A1B22}      % headings / dark surfaces
\definecolor{slink}{HTML}{26272F}       % body text
\definecolor{slaccent}{HTML}{E11D48}    % accent (crimson)
\definecolor{slaccentdk}{HTML}{BE123C}  % accent (text / links)
\definecolor{slsoft}{HTML}{5E5F6B}      % muted text
\definecolor{slfaint}{HTML}{9091A0}     % faint labels
\definecolor{sltint}{HTML}{EAE8EC}      % tinted surfaces
\definecolor{slrule}{HTML}{DFDDE3}      % hairlines
\definecolor{slborder}{HTML}{D4D2DA}    % card borders

% ── Headings: navy, sans-bold, with a small accent square ────────
\setkomafont{section}{\sffamily\bfseries\Large\color{slhead}}
\setkomafont{subsection}{\sffamily\bfseries\large\color{slhead}}
\setkomafont{subsubsection}{\sffamily\bfseries\normalsize\color{slhead}}
\setkomafont{title}{\sffamily\bfseries\color{slhead}}
\setkomafont{subtitle}{\sffamily\color{slsoft}}

% accent square before the (numbered) section heading, like the site's h2 marker
\renewcommand*{\sectionformat}{%
  \raisebox{0.06ex}{\textcolor{slaccent}{\rule{0.62em}{0.62em}}}\enspace
  \thesection\autodot\enskip}
\renewcommand*{\subsectionformat}{\thesubsection\autodot\enskip}

% ── Callouts: grey instead of Quarto's default blue ──────────────
% Quarto defines these colours later in its own preamble, so override
% them at \begin{document} (icon, frame, left rule and faint title tint).
\AtBeginDocument{%
  \definecolor{quarto-callout-note-color}{HTML}{5E5F6B}%       icon + title tint (slsoft)
  \definecolor{quarto-callout-note-color-frame}{HTML}{5E5F6B}% frame + left rule (slsoft)
}

% ── Links: accent-coloured (matches the site) ────────────────────
% Deferred: hyperref is loaded later in the preamble than this file.
\AtBeginDocument{\hypersetup{colorlinks=true, linkcolor=slaccentdk, urlcolor=slaccentdk, citecolor=slaccentdk}}

% ── Code: wrap long lines instead of overflowing the margin ──────
\usepackage{fvextra}
\AtBeginDocument{\fvset{breaklines=true, breakanywhere=true}}
